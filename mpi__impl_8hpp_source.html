112 template <class T>
113 void gather(const array_view<T, 1, row_major> &iobuff,
114  int root,
115  const MPI_Comm &comm) {
116  assert(zisa::mpi::test_intra(comm));
117  auto rank = zisa::mpi::rank(comm);
118  auto size = zisa::mpi::size(comm);
119 
120  auto ptr = (void *)raw_ptr(iobuff);
121  auto n_elements
122  = (rank == root ? iobuff.size() / int_t(size) : iobuff.size());
123  auto n_bytes = integer_cast<int>(n_elements * sizeof(iobuff[0]));
124 
125  if (rank == root) {
126  auto code = MPI_Gather(
127  MPI_IN_PLACE, 0, MPI_DATATYPE_NULL, ptr, n_bytes, MPI_BYTE, root, comm);
128  LOG_ERR_IF(code != MPI_SUCCESS, "Root failed MPI_Allgather.");
129  } else {
130  auto code = MPI_Gather(
131  ptr, n_bytes, MPI_BYTE, nullptr, 0, MPI_DATATYPE_NULL, root, comm);
132  LOG_ERR_IF(code != MPI_SUCCESS, "Worker failed MPI_Allgather.");
133  }
134 }
